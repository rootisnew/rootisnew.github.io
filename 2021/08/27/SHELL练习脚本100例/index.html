#!/bin/bash
usage(){
    echo "Usage:$0 url"
    exit 1
}
checkurl(){                                                                                                                                                                                                                                      
    local num=`curl -I -m 5 -s -w "%{http_code}\n" -o /dev/null $1 'egrep "(200'301'302)"'wc -l`
    if [ $num -eq 1 ]
    then
         echo "$1  (Website It's ok)"
    else
         echo "$1i (Website It's failed)"
    fi
}
main(){
    if [ $# -ne 1 ]
    then
        usage
    fi
    checkurl $1
}
main $*
