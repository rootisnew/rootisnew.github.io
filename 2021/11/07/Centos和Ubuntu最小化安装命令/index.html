删除一个文件里面的所有空格

#grep
grep -v "^$" a.txt
grep [^$] ip1 
#sed
sed '/^$/d' data.txt
sed '/^\s*$/d' data.txt 
#awk
awk NF data.txt
awk '!/^$/' data.txt
#tr
tr -s '\n' < data.txt
